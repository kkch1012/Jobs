# 크론탭 설정 예시
# 매일 아침 8시에 유사도 계산 배치 스크립트 실행

# 크론탭 편집: crontab -e
# 아래 라인 추가:

# 매일 아침 8시에 유사도 계산 배치 실행
0 8 * * * cd /path/to/your/project/backend && python run_similarity_batch.py >> /var/log/similarity_batch.log 2>&1

# 또는 환경변수와 함께 실행
0 8 * * * cd /path/to/your/project/backend && /path/to/venv/bin/python run_similarity_batch.py >> /var/log/similarity_batch.log 2>&1
